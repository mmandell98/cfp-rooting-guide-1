'use client';export type Objective = 'Make CFP' | 'Top 12 Seed' | 'Win Conference';const OBJECTIVES: Objective[] = ['Make CFP', 'Top 12 Seed', 'Win Conference'];function getInitialObjective(): Objective {  if (typeof localStorage === 'undefined') return 'Make CFP';  try {    const raw = localStorage.getItem('cfp.objective') || 'Make CFP';    return (OBJECTIVES as readonly string[]).includes(raw) ? (raw as Objective) : 'Make CFP';  } catch {    return 'Make CFP';  }}export default function Page() {  const [objective, setObjective] = React.useState<Objective>(getInitialObjective());  const message = `CFP Rooting Guide is running! Current objective: ${objective}`;  return (    <main style={{ padding: 20 }}>      <h1>{message}</h1>      <label style={{ fontWeight: 600 }}>Objective</label>      <select        value={objective}        onChange={(e) => setObjective(e.target.value as Objective)}        style={{ marginLeft: 10 }}      >        {OBJECTIVES.map((o) => (          <option key={o} value={o}>            {o}          </option>        ))}      </select>    </main>  );}
